afl初使用

2022-05-26 
sudo apt-get install gcc git make wget build-essential llvm clang
cd $HOME
git clone https://github.com/google/AFL.git && cd AFL/

/$HOME/AFL/llvm_mode/afl-clang-fast.c
---------------131~134行-------------------
#ifndef __ANDROID__
  cc_params[cc_par_cnt++] = "-mllvm";
  cc_params[cc_par_cnt++] = "-sanitizer-coverage-block-threshold=0";
#endif
-------------------------------------------
将这部分内容删掉，保存退出

make AFL_TRACE_PC=1
sudo make install
./afl-fuzz -V

cd $HOME
mkdir fuzzing_xpdf && cd fuzzing_xpdf
wget https://dl.xpdfreader.com/old/xpdf-3.02.tar.gz
tar -xvzf xpdf-3.02.tar.gz
cd xpdf-3.02/

cd /$HOME/AFL/llvm_mode
sudo apt-get install clang
export LLVM_CONFIG=`which llvm-config` && make && cd ..

export CC=/$HOME/AFL/afl-clang-fast
export CXX=/$HOME/AFL/afl-clang-fast++
cd $HOME/fuzzing_xpdf/xpdf-3.02
./configure --prefix="$HOME/fuzzing_xpdf/install/"
make
make install

cd $HOME/fuzzing_xpdf/install/bin
strings ./pdftotext | grep __sanitizer_cov

cd $HOME/fuzzing_xpdf
mkdir pdf_examples && cd pdf_examples
wget http://www.africau.edu/images/default/sample.pdf
